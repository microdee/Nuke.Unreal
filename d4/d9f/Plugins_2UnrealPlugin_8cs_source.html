104 /// <summary>
105 /// <para>
106 /// Get a `PluginInfo` instance from the path to its `.uplugin` file, or a file/folder
107 /// belonging to that plugin (in its subtree of files and folders). If a PluginInfo
108 /// doesn't exist yet, one will be created given that the associated plugin exists.
109 /// </para>
110 /// <para>
111 /// For example:
112 /// </para>
113 /// <code>
114 /// // In a local `MyPlugin.nuke.cs`
115 /// PluginInfo.Get(this.ScriptFolder()); // Gets or creates a `PluginInfo` about `MyPlugin.uplugin`
116 /// </code>
117 /// </summary>
118 /// <remarks>
119 /// Referring to non-existing plugins is a runtime error.
120 /// </remarks>
121 /// <param name="from"></param>
122 /// <returns></returns>
123 public static UnrealPlugin Get(AbsolutePath from)
124 {
125 var uplugin = from.FileExists() && from.HasExtension(".uplugin") ? from : from.GetOwningPlugin();
126 Assert.NotNull(uplugin, $"Given plugin context {from} didn't contain an Unreal plugin");
127 if (!Instances.ContainsKey(uplugin!))
128 {
129 Log.Debug("Handling plugin: {0}", uplugin!.NameWithoutExtension);
130 Instances.Add(uplugin!, new (uplugin!));
131 }
132 return Instances[uplugin!];
133 }

135 internal UnrealPlugin(AbsolutePath uplugin)
136 {
137 PluginPath = uplugin;
138 Descriptor = uplugin.ReadJson<PluginDescriptor>(Unreal.JsonReadSettings);
139 if (FilterPluginIni.FileExists())
140 {
141 Log.Debug("Considering FilterPlugin.ini of {0}", Name);
142 var inFiles = FilterPluginIni.ReadAllLines()
143 .Where(l => !l.Contains("[FilterPlugin]"))
144 .Where(l => !string.IsNullOrWhiteSpace(l))
145 .Where(l => !l.StartsWith(';'))
146 .Select(l => (RelativePath)l.Trim())
147 .DistinctBy(RelativePathDistinction)
148 ;
149 AddExplicitPluginFiles(inFiles);
150 }

272 {

275 var configPath = FilterPluginIni;
276 Log.Debug("Generating FilterPlugin.ini: {0}", configPath);
277
278 var lines = _explicitPluginFiles
279 .Select(f => ("/" + f.ToString()).Replace("//", "/"))
280 .Distinct()
281 ;
282 if (!lines.IsEmpty())
283 configPath.WriteAllLines(lines.Prepend("[FilterPlugin]"));
284 else
285 Log.Debug("There were no files to list in FilterPlugin.ini");

317 private bool InjectBinaryPlumbing(out PluginDescriptor result)
318 {
319 var plumbingRootRelative = (RelativePath) "Source" / "ThirdParty" / "BinaryPlumbing";
320 var plumbingRoot = Folder / plumbingRootRelative;
321 if (!plumbingRoot.DirectoryExists())
322 {
323 result = Descriptor;
324 return false;
325 }
326
327 Log.Information("{0} seemingly requires binary plumbing", Name);
328 Log.Debug("Because {0} exists", plumbingRoot);
329
330 result = Descriptor with {
331 PreBuildSteps = Descriptor.PreBuildSteps?.ToDictionary() ?? []
332 };
333 result.PreBuildSteps.EnsureDevelopmentPlatforms();
334
335 foreach (var platform in UnrealPlatform.DevelopmentPlatforms)
336 {
337 result.PreBuildSteps[platform] = [..
338 result.PreBuildSteps[platform]
339 .FilterBuildStepBlock("BinaryPlumbing")
340 .StartBuildStepBlock("BinaryPlumbing"),
341 $"echo Plumbing runtime dependencies for {Name} plugin",
342 ];
343 }
344
345 foreach (var subdir in plumbingRoot.GetDirectories())
346 {
347 Log.Debug("Plumbing from {0}", subdir);
348
349 result.PreBuildSteps[UnrealPlatform.Win64].AddRange([
350 $"echo Copying {subdir.Name} runtime dependencies",
351 $"robocopy /s /v /njh /njs /np /ndl \"$(PluginDir)\\{plumbingRootRelative.ToWinRelativePath()}\\{subdir.Name}\" \"$(PluginDir)\\Binaries\"",
352 "if %ERRORLEVEL% LSS 8 (exit /B 0) else (exit /B %ERRORLEVEL%)"
353 ]);
354
355 result.PreBuildSteps[UnrealPlatform.Linux].AddRange([
356 $"echo Copying {subdir.Name} runtime dependencies",
357 $"cp -vnpR \"$(PluginDir)/{plumbingRootRelative.ToUnixRelativePath()}/{subdir.Name}\" \"$(PluginDir)/Binaries\"",
358 "chmod -R +x \"$(PluginDir)/Binaries\""
359 ]);
360
361 result.PreBuildSteps[UnrealPlatform.Mac].AddRange([
362 $"echo Copying {subdir.Name} runtime dependencies",
363 $"cp -vnpR \"$(PluginDir)/{plumbingRootRelative.ToUnixRelativePath()}/{subdir.Name}\" \"$(PluginDir)/Binaries\"",
364 "chmod -R +x \"$(PluginDir)/Binaries\""
365 ]);
366 }
367
368 return true;
369 }

371 /// <summary>
372 /// Create a copy of this plugin which can be distributed to other developers or other tools
373 /// who shouldn't require extra non-unreal related steps to work with it.
374 /// </summary>
375 /// <param name="build"></param>
376 /// <param name="options">Optional arguments for distribution</param>
377 /// <param name="pretend">
378 /// Do not have side effects on files, just return a list of files which may be affected
379 /// by this operation.
380 /// </param>
381 /// <returns>
382 /// <list type="bullet">
383 /// <item>result: List of files which has been copied</item>
384 /// <item>output: Output folder of distribution</item>
385 /// </list>
386 /// </returns>
387 public (IEnumerable<ImportedItem> result, AbsolutePath output) DistributeSource(
388 UnrealBuild build,
389 PluginDistributionOptions? options = null,
390 bool pretend = false
391 ) {
392 var outFolder = GetDistributionOutput(build, options);
393 options ??= _distributionOptionsCache;
394
395 if (options.GenerateFilterPluginIni && !pretend)
396 GenerateFilterPluginIni(build);
397
398 var result = build.ImportFolder((Folder, outFolder, "PluginFiles.yml"), new(
399 UseSubfolder: false,
400 ForceCopyLinks: true,
401 Pretend: pretend,
402 AddToMain: [_filterExportManifest, new()
403 {
404 Copy = {
405 new() { File = "Content/**/*" , Not = { "PluginFiles.yml" }},
406 new() { File = "Shaders/**/*" , Not = { "PluginFiles.yml" }},
407 new() { File = "Source/**/*" , Not = { "PluginFiles.yml" }},
408 new() { File = "Resources/**/*" , Not = { "PluginFiles.yml" }},
409 new() { File = "Tests/**/*" , Not = { "PluginFiles.yml" }},
410 new() { File = "*.uplugin" , Not = { "PluginFiles.yml" }},
411 },
412 }]
413 ));
414
415 var outUPlugin = outFolder / PluginPath.Name;
416
417 if (!pretend && options.UPluginAssociateEngineVersion)
418 {
419 Descriptor = Descriptor with { EngineVersion = Unreal.Version(build).VersionMinor };
420 Unreal.WriteJson(Descriptor, outUPlugin);
421 }
422
423 if (!pretend && InjectBinaryPlumbing(out var newDescriptor))
424 {
425 Unreal.WriteJson(newDescriptor, outUPlugin);
426 }
427
428 return (result.WithFilesExpanded(), outFolder);
429 }

431 /// <summary>
432 /// Make a prebuilt release of this plugin for end-users. Globally set UAT and UBT arguments
433 /// are used from the input UnrealBuild
434 /// </summary>
435 /// <param name="build"></param>
436 /// <param name="uatConfig">Configurator for UAT</param>
437 /// <param name="ubtConfig">Configurator for UBT (only when UBT method is used)</param>
438 /// <param name="buildOptions">Optional arguments for packaging</param>
439 /// <param name="distOptions">Optional arguments for distribution</param>
440 /// <returns>Output folder of the packaged plugin</returns>
441 public AbsolutePath BuildPlugin(
442 UnrealBuild build,
443 Func<UatConfig, UatConfig>? uatConfig = null,
444 Func<UbtConfig, UbtConfig>? ubtConfig = null,
445 PluginBuildOptions? buildOptions = null,
446 PluginDistributionOptions? distOptions = null
447 ) {
448 var outFolder = GetBuildOutput(build, buildOptions);
449 buildOptions ??= _buildOptionsCache;
450 uatConfig ??= _ => _;
451 ubtConfig ??= _ => _;
452
453 var sourceFolder = Folder;
454 if (buildOptions.UseDistributedPlugin || buildOptions.Method == PluginBuildMethod.UBT)
455 {
456 var (_, distFolder) = DistributeSource(build, distOptions);
457 sourceFolder = distFolder;
458 }
459
460 var hostProjectDir = build.GetOutput() / "Plugins" / Name / "HostProject";
461 outFolder.CreateOrCleanDirectory();
462 hostProjectDir.ExistingDirectory()?.DeleteDirectory();
463
464 var platforms = (buildOptions.Platforms ?? []).Union([build.Platform]);
465
466 switch (buildOptions.Method)
467 {
468
469 case PluginBuildMethod.UAT:
470 var shortSource = sourceFolder.Shorten();
471 var shortOut = outFolder.Shorten();
472 try
473 {
474 Unreal.AutomationTool(build, _ => _
475 .BuildPlugin(_ => _
476 .Plugin(shortSource / PluginPath.Name)
477 .Package(shortOut)
478 .TargetPlatforms(string.Join('+', platforms))
479 .If(Unreal.Is4(build), _ => _
480 .VS2019()
481 )
482 .Unversioned()
483 )
484 .Apply(build.UatGlobal)
485 .Apply(uatConfig)
486 )("");
487 }
488 catch (Exception) { throw; }
489 finally
490 {
491 if (RuntimeInformation.IsOSPlatform(OSPlatform.Windows))
492 {
493 shortSource.DeleteDirectory();
494 shortOut.DeleteDirectory();
495 }
496 }
497
498 break;
499
500 case PluginBuildMethod.UBT:
501 hostProjectDir.CreateDirectory();
502 var hostPluginDir = hostProjectDir / "Plugins" / Name;
503 (hostProjectDir / "HostProject.uproject").WriteAllText(
504 $$"""
505 { "FileVersion": 3, "Plugins": [ { "Name": "{{Name}}", "Enabled": true } ] }
506 """
507 );
508 sourceFolder.Copy(hostPluginDir);
509 var shortPluginDir = hostPluginDir.Shorten();
510
511 try
512 {
513 foreach(var platform in platforms)
514 {
515 Log.Information("Building UnrealGame binaries for {0}", platform);
516 Unreal.BuildTool(build, _ => _
517 .Target("UnrealGame", platform,
518 [
519 UnrealConfig.Development,
520 UnrealConfig.Shipping
521 ])
522 .Plugin(shortPluginDir / PluginPath.Name)
523 .NoUBTMakefiles()
524 .NoHotReload()
525 .Apply(ubtConfig)
526 .Apply(build.UbtGlobal)
527 )("");
528 if (platform.IsDevelopment)
529 {
530 Log.Information("Building UnrealEditor binaries for {0}", platform);
531 Unreal.BuildTool(build, _ => _
532 .Target("UnrealEditor", platform, [UnrealConfig.Development])
533 .Plugin(shortPluginDir / PluginPath.Name)
534 .NoUBTMakefiles()
535 .NoHotReload()
536 .Apply(ubtConfig)
537 .Apply(build.UbtGlobal)
538 )("");
539 }
540 }
541 hostPluginDir.Copy(outFolder, ExistsPolicy.MergeAndOverwrite);
542 hostProjectDir.DeleteDirectory();
543 }
544 catch(Exception) { throw; }
545 finally
546 {
547 if (RuntimeInformation.IsOSPlatform(OSPlatform.Windows))
548 shortPluginDir.DeleteDirectory();
549 }
550 break;
551
552 }
553
554 return outFolder;
555 }
